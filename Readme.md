``` HTML
    <!DOCTYPE html>
    <html>
    <head>
    <style>
    .city {
            background-color: yellow;
            color: black;
            border: 2px solid black;
            margin: 20px;
            padding: 20px;
    }
    </style>
    </head>
    <body>

     <div class="city">
        <h2>Berlin</h2>
        <p>Berlin is the capital of Germany</p>
    </div> 

    <div class="city">
        <h2>Damascus</h2>

    </body>

}
```
